import pygame import sys import random # Inicializa o pygame pygame.init() # Configurações da tela WIDTH, HEIGHT = 800, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("PONG - Robson Lincon Serpe Garcia") # Cores BLACK = (0, 0, 0) WHITE = (255, 255, 255) RED = (255, 0, 0) BLUE = (0, 0, 255) # Configurações do jogo FPS = 60 PADDLE_WIDTH, PADDLE_HEIGHT = 15, 100 BALL_SIZE = 15 PADDLE_SPEED = 7 BALL_SPEED_X, BALL_SPEED_Y = 5, 5 # Pontuação score_left = 0 score_right = 0 font = pygame.font.Font(None, 74) # Cria as raquetes e a bola left_paddle = pygame.Rect(50, HEIGHT // 2 - PADDLE_HEIGHT // 2, PADDLE_WIDTH, PADDLE_HEIGHT) right_paddle = pygame.Rect(WIDTH - 50 - PADDLE_WIDTH, HEIGHT // 2 - PADDLE_HEIGHT // 2, PADDLE_WIDTH, PADDLE_HEIGHT) ball = pygame.Rect(WIDTH // 2 - BALL_SIZE // 2, HEIGHT // 2 - BALL_SIZE // 2, BALL_SIZE, BALL_SIZE) # Direção da bola ball_dx = BALL_SPEED_X * random.choice((1, -1)) ball_dy = BALL_SPEED_Y * random.choice((1, -1)) # Relógio para controlar o FPS clock = pygame.time.Clock() def reset_ball(): """Reseta a bola para o centro com direção aleatória""" ball.center = (WIDTH // 2, HEIGHT // 2) global ball_dx, ball_dy ball_dx = BALL_SPEED_X * random.choice((1, -1)) ball_dy = BALL_SPEED_Y * random.choice((1, -1)) # Loop principal do jogo while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() # Movimento das raquetes keys = pygame.key.get_pressed() if keys[pygame.K_w] and left_paddle.top > 0: left_paddle.y -= PADDLE_SPEED if keys[pygame.K_s] and left_paddle.bottom < HEIGHT: left_paddle.y +=PADDLE_SPEED if keys[pygame.K_UP] and right_paddle.top> 0: right_paddle.y -= PADDLE_SPEED if keys[pygame.K_DOWN] and right_paddle.bottom < HEIGHT: right_paddle.y +=PADDLE_SPEED # Movimento da bola ball.x +=ball_dx ball.y +=ball_dy # Colisão com as paredes superior e inferior if ball.top <=0 or ball.bottom>= HEIGHT: ball_dy *= -1 # Colisão com as raquetes if ball.colliderect(left_paddle) or ball.colliderect(right_paddle): ball_dx *= -1.1 # Aumenta a velocidade e inverte a direção ball_dy *= 1.1 # Pontuação if ball.left <= 0: score_right +=1 reset_ball() if ball.right>= WIDTH: score_left += 1 reset_ball() # Desenha tudo na tela screen.fill(BLACK) pygame.draw.rect(screen, WHITE, left_paddle) pygame.draw.rect(screen, WHITE, right_paddle) pygame.draw.ellipse(screen, RED, ball) pygame.draw.aaline(screen, WHITE, (WIDTH // 2, 0), (WIDTH // 2, HEIGHT)) # Desenha a pontuação score_text = font.render(f"{score_left} {score_right}", True, WHITE) screen.blit(score_text, (WIDTH // 2 - score_text.get_width() // 2, 20)) # Atualiza a tela pygame.display.flip() clock.tick(FPS)
